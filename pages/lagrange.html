double lagranz(double X[n], double Y[n], double t) {
double sum, prod;
sum = 0;
for (int j = 0; j < n; j++) {
prod = 1;
for (int i = 0; i < n; i++) {
if (i != j) {
prod = prod * (t - X[i]) / (X[j] - X[i]);
}
}
sum = sum + Y[j] * prod;
}
return sum;
}

double pirson(double X[n], double Y[n]) {
double sredX, sredY, dispX, dispY, rXY, covXY;
for (int j = 0; j < n; j++) {
rXY = 0;
for (int i = 0; i < n; i++) {
if (i != j) {
sredX = X[i] / n;
sredY = Y[i] / n;
dispX = pow(X[i] - X[j], 2);
dispY = pow(Y[i] - Y[j], 2);
covXY = (X[i] - X[j]) * (Y[i] - Y[j]);
}
}
rXY = covXY / sqrt(dispX * dispY);
}
return rXY;
}
