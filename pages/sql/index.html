1. Выборка данных

SELECT: используется для выбора данных из базы данных

  • SELECT * FROM имя_таблицы;

DISTINCT: отфильтровывает повторяющиеся значения и возвращает строки определенного столбца

  • SELECT DISTINCT имя_столбца;

WHERE: используется для фильтрации записей / строк

  • SELECT столбец1, столбец2 FROM имя_таблицы WHERE условие;
  • SELECT * FROM имя_таблицы WHERE условие1 AND условие2;
  • SELECT * FROM имя_таблицы WHERE условие1 OR условие2;
  • SELECT * FROM имя_таблицы WHERE NOT условие;
  • SELECT * FROM имя_таблицы WHERE условие1 AND (условие2 OR условие3);
  • SELECT * FROM имя_таблицы WHERE EXISTS (SELECT имя_столбца FROM имя_таблицы WHERE условие);

ORDER BY: используется для сортировки набора результатов в порядке возрастания или убывания

  • SELECT * FROM имя_таблицы ORDER BY столбец;
  • SELECT * FROM имя_таблицы ORDER BY столбец DESC;
  • SELECT * FROM имя_таблицы ORDER BY столбец1 ASC, столбец2 DESC;

SELECT TOP: используется для указания количества записей, возвращаемых из верхней части таблицы

  • SELECT TOP number имена_столбцов FROM имя_таблицы WHERE условие;
  • SELECT TOP percent имена_столбцов FROM имя_таблицы WHERE условие;
  • Не все системы баз данных поддерживают SELECT TOP. Эквивалентом MySQL является LIMIT
  • SELECT имена_столбцов FROM имя_таблицы LIMIT смещение, кол-во;

LIKE: оператор, используемый в предложении WHERE для поиска определенного шаблона в столбце

  • % (Знак процента) является символом подстановки, который представляет ноль, один или несколько символов
  • _ (подчеркивание) - это символ подстановки, который представляет один символ
  • SELECT имена_столбцов FROM имя_таблицы WHERE имя_столбца LIKE шаблон;
  • LIKE ‘a%’ (Найти любые значения, начинающиеся с «а»)
  • LIKE ‘%a’ (найти любые значения, которые заканчиваются на «а»)
  • LIKE ‘%or%’ (найти любые значения, которые имеют «or» в любой позиции)
  • LIKE ‘_r%’ (найти любые значения, которые имеют «r» во второй позиции)
  • LIKE ‘a_%_%’ (найти любые значения, которые начинаются с «а» и имеют длину не менее 3 символов)
  • LIKE ‘[a-c]%’ (найти любые значения, начинающиеся с «a», «b» или «c»)

IN: оператор, который позволяет указать несколько значений в предложении WHERE

  • По сути, оператор IN является сокращением для нескольких условий OR
  • SELECT имена_столбцов FROM имя_таблицы WHERE имя_столбца IN (значение1, значение2, …);
  • SELECT имена_столбцов FROM имя_таблицы WHERE имя_столбца IN (SELECT STATEMENT);

BETWEEN: оператор выбирающий значения в заданном диапазоне включительно

  • SELECT имена_столбцов FROM имя_таблицы WHERE имя_столбца BETWEEN значение1 AND значение2;
  • SELECT * FROM Продукты WHERE (имя_столбца1 BETWEEN значение1 AND значение2) AND NOT имя_столбца2 IN (значение3, значение4);
  • SELECT * FROM Продукты WHERE имя_столбца BETWEEN #01/07/1999# AND #03/12/1999#;
